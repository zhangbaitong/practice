根据Dockerfile生成image
docker build -t="centos7-gearman-php-chen" .


启动容器
docker  run -d  -p 4730:4730 --name gearman-php-chen -v /www/business_card/backend_service:/www/business_card/backend_service centos7-gearman-php-chen


启动后台程序
cd /www/business_card/backend_service
chmod +x start.sh
./start.sh start
